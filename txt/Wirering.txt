Clock:
CLK D2 - CTR D2

Controller line: 
CTR D4 <- (PC, MAR, RAM, IR, A, ALU, B, OUT) D3

Bus line:
  PC D6 - MAR D5 - IR D6 
  RAM D6 - A D5 - ALU D6 - B D5
  A D6 - OUT D5

A-ALU-B part:
  A D10 - ALU D9
  B D8 - ALU D7

HardwareSerial:
  CTR D0 <- IR D1
  CTR D1 <- IR D0

  MAR D0 <- RAM D1
  MAR D1 <- RAM D0

  OUT D0 <- Display D1
  Display D1 <- OUT D0
